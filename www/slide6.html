import javax.sound.sampled.AudioInputStream;

public class Song {
    private int bytesPerFrame; //= audioInputStream.getFormat().getFrameSize();
    private long fileSize; //= audioInputStream.getFrameLength() * bytesPerFrame;
    private float bytesFrame; //= (bytesPerFrame * audioInputStream.getFormat().getFrameRate());
    private float songTimeLength; //= (fileSize / bytesFrame);
    private int bytesPerSecond;// = (int) ((fileSize - 44) / songTimeLength);
    private int chunkSize; //= size of FFT chunk
    private int numChunks;//= (int) fileSize / numBytes      
    
    
    public Song(AudioInputStream ais) {
    	this.bytesPerFrame = ais.getFormat().getFrameSize();
    	this.fileSize = ais.getFrameLength() * this.bytesPerFrame;
    	this.bytesFrame = this.bytesPerFrame * ais.getFormat().getFrameRate();
    	this.songTimeLength = this.fileSize / this.bytesFrame;
    	this.bytesPerSecond = (int)((this.fileSize - 44) / this.songTimeLength);
    	this.chunkSize = (int)(1024);
    	this.numChunks = (int)(this.fileSize) / this.chunkSize;
    }
